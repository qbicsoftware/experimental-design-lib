110   public List<Study> listStudies(File file) {
111     error = null;
112     final String configDir = resolveConfigurationFilesPath();
113     importer = new ISAtabFilesImporter(configDir);
114     isatabParentDir = file.toString();
115     try {
116       importer.importFile(isatabParentDir);
117       investigation = importer.getInvestigation();
118     } catch (NullPointerException e) {
119       error = "Investigation file not found or not the right format.";
120     }
121 
122     for (ISAFileErrorReport report : importer.getMessages()) {
123       for (ErrorMessage message : report.getMessages()) {
124         error = mapError(message.getMessage());
125         log.error(message.getErrorLevel().toString() + " > " + message.getMessage());
126       }
127     }
128     List<Study> res = new ArrayList<Study>();
129     if (error == null)
130       res.addAll(investigation.getStudies().values());
131     return res;
132   }

134   private String mapError(String message) {
135     String res = message;
136     if (message.startsWith("Investigation file does not exist"))
137       res = "No Investigation file could be found.";
138     if (message.contains(
139         "Please ensure that the file exists within the folder and that the name referred to in the investigation file is correct!")) {
140       String file = message.split(" was not found")[0];
141       if (file.contains("/")) {
142         String[] splt = file.split("/");
143         file = splt[splt.length - 1];
144       } else {
145         file = file.split("The file ")[1];
146       }
147       res = file
148           + " not found. Please ensure that the file exists within the folder and that the name referred to in the investigation file is correct.";
149     }
150     return res;
151   }

320   private Set<String> getFactorsForSources(Study study) {
321     Set<String> res = new HashSet<>();
322     for (Factor f : study.getFactors()) {
323       Map<String, String> sourcesToValue = new HashMap<>();
324       String label = f.getFactorName();
325       res.add(label);
326       int sourceCol = findStudyColumnID(study, "Source Name");
327       Object[][] matrix = study.getStudySampleDataMatrix();
328       for (int rowID = 1; rowID < matrix.length; rowID++) {
329         String sourceID = (String) matrix[rowID][sourceCol];
330         String colLabel = "Factor Value[" + label + "]";
331         int factorCol = findStudyColumnID(study, colLabel);
332         if (factorCol != -1) {
333           String factorVal = (String) matrix[rowID][factorCol];
334           String entry = sourcesToValue.get(sourceID);
335           // not source factor, as there are different entries
336           if (entry != null && !entry.equals(factorVal)) {
337             res.remove(label);
338             break;
339           } else {
340             sourcesToValue.put(sourceID, factorVal);
341           }
342         }
343       }
344     }
345     return res;
346   }

348   private String resolveConfigurationFilesPath() {
349     if (CONFIG_PATH != null)
350       return CONFIG_PATH;
351     for (int i = 0; i < DEFAULT_CONFIG_PATHS.length; i++) {
352       final File possibleConfigFolder = DEFAULT_CONFIG_PATHS[i].toFile();
353       if (possibleConfigFolder.exists() && possibleConfigFolder.isDirectory()) {
354         return DEFAULT_CONFIG_PATHS[i].toString();
355       } else {
356         log.info(String.format("Configuration files not found in folder %s",
357             DEFAULT_CONFIG_PATHS[i].toString()));
358       }
359     }
360     // TODO: change for ApplicationException, one of QBiC's "generic" runtime exceptions
361     throw new RuntimeException(
362         "Required configuration files were not found at any of the default folders.");
363   }

402   private void createGraphSummariesForRow(List<TSVSampleBean> levels, int oldID) {
403     // nodeID *= levels.size();
404     // create summary for this each node based on each experimental factor as well as "none"
405     for (String label : nodesForFactorPerLabel.keySet()) {
406       SampleSummary currentSummary = null;
407       // source - extract - analyte (at least the first must exist) - create summaries each
408       for (int level = 0; level < levels.size(); level++) {
409         TSVSampleBean s = levels.get(level);
410         // find out if this sample has children or not
411         int next = level + 1;
412         boolean leaf = levels.size() == next || levels.get(next) == null;
413         // sample on this level does exist
414         if (s != null) {
415           // int id = nodeID * next + 1;
416           nodeID++;
417           String idCat = "" + next + nodeID;
418           int id = Integer.parseInt(idCat);
419           Set<SampleSummary> parentSummaries = new LinkedHashSet<SampleSummary>();
420           if (currentSummary != null)
421             parentSummaries.add(currentSummary);
422           currentSummary = createNodeSummary(s, parentSummaries, label, id, leaf);
423           // check for hashcode and add current sample s if node exists and doesn't contain code yet
424           boolean exists = false;
425           for (SampleSummary oldNode : nodesForFactorPerLabel.get(label)) {
426             if (oldNode.equals(currentSummary)) {
427               for (String code : currentSummary.getCodes()) {
428                 exists = oldNode.getCodes().contains(code);
429               }
430               if (!exists)
431                 oldNode.addSample(s);
432               currentSummary = oldNode;
433             }
434           }
435           // adds node if not already contained in set
436           Set<SampleSummary> theseNodes = nodesForFactorPerLabel.get(label);
437           theseNodes.add(currentSummary);
438           nodesForFactorPerLabel.put(label, theseNodes);
439           // add this id to parents' child ids
440           for (SampleSummary parentSummary : parentSummaries) {
441             parentSummary.addChildID(currentSummary.getId());
442           }
443         }
444       }
445     }
446   }

448   // new "sample to bucket" function, creates new summaries from sample metadata in reference to
449   // parent summaries and experimental factor
450   private SampleSummary createNodeSummary(TSVSampleBean s, Set<SampleSummary> parents, String label,
451       int currentID, boolean leaf) {
452     // name: should be the visible discriminating factor between nodes
453     // 1. contains the source, if the source is not the selected factor (e.g. tissues)
454     // 2. contains the selected factor's value, except
455     // a) if parent sample has the same factor value
456     // b) if it has no factor
457     // factor: the current selected factor object. If none exists, parents' sources are used.
458 
459     // the name alone is not enough to discriminate between different nodes! (e.g. different parent
460     // nodes, same child node name)
461     SampleType type = s.getType();
462     String source = "unknown";
463     Map<String, Object> props = s.getMetadata();
464 
465     Property factor = getFactorOfSampleOrNull((List<Property>) props.get("Factors"), label);
466     boolean newFactor = true;
467     Set<String> parentSources = new HashSet<String>();
468     Set<Integer> parentIDs = new HashSet<Integer>();
469     for (SampleSummary parentSum : parents) {
470       parentIDs.add(parentSum.getId());
471       String factorVal = parentSum.getFactorValue();
472       if (factorVal != null && !factorVal.isEmpty()) {
473         newFactor = false;
474       }
475       parentSources.add(parentSum.getSource());
476     }
477     if (factor == null) {
478       factor = new Property("parents", String.join("+", parentSources), PropertyType.Factor);// TODO
479                                                                                              // makes
480                                                                                              // sense?
481       newFactor = false;
482     }
483     String value = "";
484     if (newFactor)
485       value = factor.getValue();
486     switch (type) {
487       case Q_BIOLOGICAL_ENTITY:
488         source = (String) props.get("Q_NCBI_ORGANISM");
489         value = source + " " + value;
490         if (value.isEmpty())
491           value = "unspecified source";
492         break;
493       case Q_BIOLOGICAL_SAMPLE:
494         source = (String) props.get("Q_PRIMARY_TISSUE");
495         if (!newFactor || source.equals(value)) {
496           value = source;
497         } else {
498           value = source + " " + value;
499         }
500         if (value.isEmpty())
501           value = "unspecified extract";
502         break;
503       case Q_TEST_SAMPLE:
504         source = (String) props.get("Q_SAMPLE_TYPE");
505         value = source + " " + value;
506         break;
507     }
508     return new SampleSummary(currentID, parentIDs, new ArrayList<ISampleBean>(Arrays.asList(s)),
509         factor.getValue(), tryShortenName(value, s).trim(), type.toString(), leaf);
510   }

603       String sourceID = (String) matrix[rowID][sourceCol];
604       String sampleID = (String) matrix[rowID][sampleCol];
605       List<Property> factors = new ArrayList<Property>();
606       List<Property> sourceFactors = new ArrayList<Property>();
607       Set<String> unknownUnits = new HashSet<String>();
608       for (String factorLabel : nodesForFactorPerLabel.keySet()) {
609         String colLabel = "Factor Value[" + factorLabel + "]";
610         int factorCol = findStudyColumnID(study, colLabel);
611         if (factorCol != -1) {
612           Property factor = null;
613           String factorVal = (String) matrix[rowID][factorCol];
614           if (studyFactorHasUnit(study, colLabel)) {
615             String unitBlock = (String) matrix[rowID][factorCol + 1];
616             Unit unit = null;
617             try {
618               unitBlock = removeOntologyPrefix(unitBlock);
619               unit = Unit.fromString(unitBlock);
620             } catch (IllegalArgumentException e) {
621               unknownUnits.add(e.toString());
622               // TODO unknown units
623               // factorVal += " " + unitBlock;
624               unit = Unit.Arbitrary_Unit;
625             } finally {
626               factor = new Property(factorLabel, factorVal, unit, PropertyType.Factor);
627             }
628           } else {
629             factor = new Property(factorLabel, factorVal, PropertyType.Factor);
630           }
631           factors.add(factor);
632           if (sourceFactorLabels.contains(factorLabel)) {
633             sourceFactors.add(factor);
634           }
635         }
636       }
637       if (!unknownUnits.isEmpty()) {
638         log.warn(unknownUnits);
639         log.warn("Units have been replaced by \"Arbitrary Unit\"");
640       }
641       // these will be filled below, if they don't exist
642       TSVSampleBean eSample = sampleIDToSample.get(sampleID);
643       TSVSampleBean sSample = sourceIDToSample.get(sourceID);
644       if (!sampleIDToSample.containsKey(sampleID)) {
645         Map<String, Object> metadata = new HashMap<String, Object>();
646         metadata.put("Factors", factors);
647         eSample = new TSVSampleBean(sampleID, SampleType.Q_BIOLOGICAL_SAMPLE, sampleID, metadata);
648         eSample.addProperty("Q_PRIMARY_TISSUE", tissue);
649         eSample.addProperty("Q_EXTERNALDB_ID", sampleID);
650         sampleIDToSample.put(sampleID, eSample);
651         eSample.addParentID(sourceID);
652 
653       }
654       if (!sourceIDToSample.containsKey(sourceID)) {
655         // sampleID++;
656         Map<String, Object> metadata = new HashMap<String, Object>();
657         metadata.put("Factors", sourceFactors);
658         sSample = new TSVSampleBean(sourceID, SampleType.Q_BIOLOGICAL_ENTITY, sourceID, metadata);
659         sSample.addProperty("Q_NCBI_ORGANISM", organism);
660         sSample.addProperty("Q_EXTERNALDB_ID", sourceID);
661         sourceIDToSample.put(sourceID, sSample);
662       }
663     }
664 
665     // for each assay find entities by Names and connect them via Sample Names to existing tissue
666     // samples
667     int unknownExtractID = 0;
668     int uniqueEntityID = 0;
669     for (String ass : study.getAssays().keySet()) {
670       Assay assay = study.getAssays().get(ass);
671       int assaySampleIDCol = findAssayColumnID(assay, "Sample Name");
672       int assayExtractIDCol = findAssayColumnID(assay, "Extract Name");
673       // Analyte
674       String endpoint = assay.getMeasurementEndpoint();
675       technologyTypes.add(new TechnologyType(endpoint));
676       String analyte = getAnalyteFromMeasureEndpoint(endpoint);
677       analyteSet.add(analyte);
678 
679       Object[][] assayMatrix = assay.getAssayDataMatrix();
680       for (int rowID = 1; rowID < assayMatrix.length; rowID++) {
681         String sampleID = (String) assayMatrix[rowID][assaySampleIDCol];
682         unknownExtractID++;
683         String extractID = Integer.toString(unknownExtractID);
684         if (assayExtractIDCol != -1) {
685           extractID = extractID + "-" + (String) assayMatrix[rowID][assayExtractIDCol];
686         }
687         TSVSampleBean eSample = sampleIDToSample.get(sampleID);
688         Map<String, Object> metadata = new HashMap<String, Object>();
689         metadata.put("Factors", eSample.getMetadata().get("Factors"));
690 
691         TSVSampleBean tSample = new TSVSampleBean(extractID, SampleType.Q_TEST_SAMPLE, extractID, metadata);
692 
693         tSample.addProperty("Q_SAMPLE_TYPE", analyte);
694         tSample.addProperty("Q_EXTERNALDB_ID", extractID);
695         analyteIDToSample.put(extractID, tSample);
696         tSample.addParentID(sampleID);
697 
698         List<TSVSampleBean> sampleRow = new ArrayList<TSVSampleBean>(
699             Arrays.asList(sourceIDToSample.get(eSample.getParentIDs().get(0)),
700                 sampleIDToSample.get(sampleID), tSample));
701         // if (analytesIncluded)
702         // sampleRow.add(analyteIDToSample.get(analyteID));
703         uniqueEntityID++;
704         createGraphSummariesForRow(sampleRow, new Integer(uniqueEntityID));
705       }
706     }
707 
708     Map<String, List<SampleSummary>> nodeListsPerLabel = new HashMap<String, List<SampleSummary>>();
709     for (String label : nodesForFactorPerLabel.keySet()) {
710       nodeListsPerLabel.put(label, new ArrayList<SampleSummary>(nodesForFactorPerLabel.get(label)));
711     }
712     currentGraphStructure = new StructuredExperiment(nodeListsPerLabel, study);
713 
714     res.addAll(sourceIDToSample.values());
715     res.addAll(sampleIDToSample.values());
716     res.addAll(analyteIDToSample.values());
717     return res;
718   }
