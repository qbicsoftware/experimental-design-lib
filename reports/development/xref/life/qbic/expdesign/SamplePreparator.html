46     * Reads in a TSV File containing samples for openBIS registration and their metadata.
47     * 
48     * @param file A TSV File that must contain Identifier, space, project, experiment and sample type
49     *        of each sample and optionally the parents of each sample. Additional metadata columns
50     *        must be known to openBIS and fit to the sample type in question.
51     * @param designType
52     * @throws IOException
53     * @throws JAXBException
54     */
55    public boolean processTSV(File file, IExperimentalDesignReader reader, boolean parseGraph)
56        throws IOException, JAXBException {
57      this.reader = reader;
58  
59      List<ISampleBean> rawSamps = reader.readSamples(file, parseGraph);
60      if (reader instanceof QBiCDesignReader) {
61        QBiCDesignReaderfe/qbic/expdesign/io/QBiCDesignReader.html#QBiCDesignReader">QBiCDesignReader qReader = (QBiCDesignReader) reader;
62        projectinfo = new ProjectInfo(qReader.getSpace(), qReader.getProject(),
63            qReader.getDescription(), qReader.getSecondaryName(), qReader.isPilot(),
64            qReader.getInvestigator(), qReader.getContact(), qReader.getManager());
65      }
66      if (reader.getError() != null)
67        return false;
68  
69      // map to save children, used to find out more about project structure to present in summary
70      // beans
71      Map<String, List<ISampleBean>> sampleToChildrenMap = new HashMap<String, List<ISampleBean>>();
72      idsToSamples = new HashMap<String, ISampleBean>();
73      experimentalDesignXML =
74          ParserHelpers.samplesWithMetadataToExperimentalFactorStructure(rawSamps);
75      for (ISampleBean b : rawSamps) {
76        idsToSamples.put(b.getCode(), b);
77        // translate tsv presentation of special metadata to xml
78        // ParserHelpers.fixXMLProps(b.getMetadata());
79  
80        // fill children map
81        for (String parent : b.getParentIDs()) {
82          if (sampleToChildrenMap.containsKey(parent))
83            sampleToChildrenMap.get(parent).add(b);
84          else
85            sampleToChildrenMap.put(parent, new ArrayList<ISampleBean>(Arrays.asList(b)));
86        }
87      }
88      processTSV(rawSamps, sampleToChildrenMap);
89      return true;
90    }

92    private void processTSV(List<ISampleBean> samples,
93        Map<String, List<ISampleBean>> sampleToChildrenMap) {
94      this.samples = samples;
95      ArrayList<SampleSummaryBean> summary = new ArrayList<SampleSummaryBean>();
96      Map<Integer, List<ISampleBean>> levels = null;
97      levels = readLevels();
98      ArrayList<Integer> numLevels = new ArrayList<Integer>(levels.keySet());
99      Collections.sort(numLevels);
100     List<List<ISampleBean>> processed = new ArrayList<List<ISampleBean>>();
101     Comparator<ISampleBean> comp = SampleTypeComparator.getInstance();
102     for (int i : numLevels) {
103       List<ISampleBean> level = new ArrayList<ISampleBean>();
104 
105       List<ISampleBean> currentLevel = levels.get(i);
106       Collections.sort(currentLevel, comp);
107 
108       SampleType type = currentLevel.get(0).getType();
109       ArrayList<ISampleBean> currList = new ArrayList<ISampleBean>();
110       for (ISampleBean s : currentLevel) {
111         if (type.equals(s.getType()))
112           currList.add(s);
113         else {
114           level.addAll(currList);
115           fetchSummary(summary, currList, i, sampleToChildrenMap);
116           currList = new ArrayList<ISampleBean>(Arrays.asList(s));
117           type = s.getType();
118         }
119       }
120       fetchSummary(summary, currList, i, sampleToChildrenMap);
121 
122       level.addAll(currList);
123       processed.add(level);
124     }
125     this.processed = processed;
126     this.summary = summary;
127   }

134   public Map<String, Map<String, Object>> transformAndReturnSpecialExperimentsOfTypeOrNull(String experimentType) {
135     if (reader.getExperimentInfos() == null)
136       return null;
137     List<Map<String, Object>> exps = reader.getExperimentInfos().get(experimentType);
138     if (exps != null) {
139       Map<String, Map<String, Object>> res = new HashMap<String, Map<String, Object>>();
140       for (Map<String, Object> exp : exps) {
141         String code = (String) exp.get("Code");
142         exp.remove("Code");
143         res.put(code, exp);
144       }
145       return res;
146     } else
147       return null;
148   }

199       for (ISampleBean b : level)
200         newLevel.add(b.copy());
201       res.add(newLevel);
202     }
203     return res;

214   public String toGraphML() {
215     if (processed == null)
216       return "";
217     else {
218       String res = "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"no\"?> \n"
219           + "<graphml xmlns=\"http://graphml.graphdrawing.org/xmlns\" "
220           + "xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" "
221           + "xmlns:y=\"http://www.yworks.com/xml/graphml\" "
222           + "xmlns:yed=\"http://www.yworks.com/xml/yed/3\" "
223           + "xsi:schemaLocation=\"http://graphml.graphdrawing.org/xmlns "
224           + "http://www.yworks.com/xml/schema/graphml/1.1/ygraphml.xsd\">"
225           + "<key for=\"node\" id=\"d1\" yfiles.type=\"nodegraphics\"/> \n"
226           + "<key for=\"edge\" id=\"d2\" yfiles.type=\"edgegraphics\"/> \n"
227           + "<graph id=\"Experimental Design\" edgedefault=\"undirected\"> \n";
228       int e = 1;
229       for (List<ISampleBean> lis : processed) {
230         for (ISampleBean b : lis) {
231           String code = b.getCode();
232           res += "<node id=\"" + code + "\"> \n" + "<data key=\"d1\"> \n" + "<y:ShapeNode> \n"
233               + "<y:Geometry width=\"105\"/>" + "<y:NodeLabel>" + code + " \n" + "</y:NodeLabel> \n"
234               + "</y:ShapeNode> \n" + "</data> \n" + "</node> \n";
235           if (b.hasParents()) {
236             List<String> parents = b.getParentIDs();
237             for (String p : parents) {
238               res += "<edge id=\"" + e + "\" source=\"" + code + "\" target=\"" + p + "\"> \n"
239                   + "<data key=\"d2\"> \n" + "<y:PolyLineEdge> \n"
240                   + "<y:Arrows source=\"none\" target=\"none\"/> \n" + "</y:PolyLineEdge> \n"
241                   + "</data> \n" + "</edge> \n";
242               e++;
243             }
244           }
245         }
246       }
247       res += "</graph> \n" + "</graphml>";
248       return res;
249     }

252   // Helper for creating a summary (as list of SampleSummaryBeans) of the processed tsv file
253   private void fetchSummary(ArrayList<SampleSummaryBean> summary, ArrayList<ISampleBean> currList,
254       int i, Map<String, List<ISampleBean>> sampleToChildrenMap) {
255     boolean isPartOfSplit = false;
256     boolean pool = false;
257     Set<String> sampleContent = new HashSet<String>();
258     for (ISampleBean s : currList) {
259       String queryType = null;
260       switch (s.getType()) {
261         case Q_BIOLOGICAL_ENTITY:
262           queryType = "Q_NCBI_ORGANISM";
263           break;
264         case Q_BIOLOGICAL_SAMPLE:
265           queryType = "Q_PRIMARY_TISSUE";
266           break;
267         case Q_TEST_SAMPLE:
268           queryType = "Q_SAMPLE_TYPE";
269           break;
270         default:
271           queryType = null;
272           break;
273       }
274       if (queryType != null) {
275         String content = (String) s.getMetadata().get(queryType);
276         sampleContent.add(content);
277       }
278 
279       List<String> parentCodes = s.getParentIDs();
280       if (parentCodes.size() > 1) {
281         pool = true;
282       } else {
283         SampleType type = s.getType();
284         for (String parent : parentCodes) {
285           // at least one parent leads to this child sample and at least one other child sample
286           // (sibling) of the same type
287           for (ISampleBean sibling : sampleToChildrenMap.get(parent)) {
288             if (!s.getCode().equals(sibling.getCode()) && sibling.getType().equals(type))
289               isPartOfSplit = true;
290           }
291         }
292 
293       }
294     }
295     summary.add(new SampleSummaryBean(currList.get(0).getType().toString(), sampleContent,
296         Integer.toString(currList.size()), pool, isPartOfSplit));
297   }

299   // expects samples to be in the right order already, with the first tier coming first and the last
300   // tier coming later
301   // new levels are created when the sample type changes or if a sample is found with a parent in an
302   // existing tier
303   private Map<Integer, List<ISampleBean>> readLevels() {
304     Map<Integer, List<ISampleBean>> knownLevels = new HashMap<Integer, List<ISampleBean>>();
305     SampleType type = samples.get(0).getType();
306     int i = 1;
307     List<ISampleBean> level = new ArrayList<ISampleBean>();
308     Set<String> ids = new HashSet<String>();
309     for (ISampleBean b : samples) {
310       boolean newLvl = false;
311       if (!b.getType().equals(type)) {// || !b.getExperiment().equals(exp)) {
312         newLvl = true;
313       } else if (b.hasParents()) {
314         for (String id : b.getParentIDs()) {
315           newLvl |= ids.contains(id);
316         }
317       }
318       if (newLvl) {
319         knownLevels.put(i, level);
320         i++;
321         type = b.getType();
322         level = new ArrayList<ISampleBean>();
323         ids = new HashSet<String>();
324       }
325       ids.add(b.getCode());
326       level.add(b);
327     }
328     knownLevels.put(i, level);
329     return knownLevels;
330   }
