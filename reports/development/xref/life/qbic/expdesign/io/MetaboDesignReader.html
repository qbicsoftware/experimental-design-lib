117   private void addValueForCategory(Map<String, Integer> headerMapping, String[] row, String cat) {
118     if (headerMapping.containsKey(cat)) {
119       String val = row[headerMapping.get(cat)];
120       if (val != null && !val.isEmpty()) {
121         for (String v : val.split(LIST_SEPARATOR)) {
122           if (parsedCategoriesToValues.containsKey(cat)) {
123             parsedCategoriesToValues.get(cat).add(v);
124           } else {
125             Set<String> set = new HashSet<String>();
126             set.add(v);
127             parsedCategoriesToValues.put(cat, set);
128           }
129         }
130       }
131     }
132   }

154   public List<ISampleBean> readSamples(File file, boolean parseGraph) throws IOException {
155     tsvByRows = new ArrayList<String>();
156     parsedCategoriesToValues = new HashMap<>();
157 
158     BufferedReader reader = new BufferedReader(new FileReader(file));
159     ArrayList<String[]> data = new ArrayList<String[]>();
160     String next;
161     int i = 0;
162     while ((next = reader.readLine()) != null) {
163       i++;
164       next = removeUTF8BOM(next);
165       tsvByRows.add(next);
166       String[] nextLine = next.split("\t", -1);// this is needed for trailing tabs
167       if (data.isEmpty() || nextLine.length == data.get(0).length) {
168         data.add(nextLine);
169       } else {
170         error = "Wrong number of columns in row " + i;
171         reader.close();
172         return null;
173       }
174     }
175     reader.close();
176 
177     String[] header = data.get(0);
178     String[] trimmedHeader = new String[header.length];
179     for (int j = 0; j < header.length; j++) {
180       trimmedHeader[j] = header[j].trim();
181     }
182     header = trimmedHeader;
183     data.remove(0);
184     // find out where the mandatory and other metadata data is
185     Map<String, Integer> headerMapping = new HashMap<>();
186     List<Integer> meta = new ArrayList<>();
187     Set<Integer> factors = new HashSet<>();
188     List<Integer> properties = new ArrayList<>();
189     List<Integer> loci = new ArrayList<>();
190 
191     ArrayList<String> found = new ArrayList<String>(Arrays.asList(header));
192     for (String col : mandatoryColumns) {
193       if (!found.contains(col)) {
194         error = "Mandatory column " + col + " not found.";
195         return null;
196       }
197     }
198     for (i = 0; i < header.length; i++) {
199       int position = mandatoryColumns.indexOf(header[i]);
200       if (position == -1)
201         position = optionalCols.indexOf(header[i]);
202       if (position > -1) {
203         headerMapping.put(header[i], i);
204         meta.add(i);
205       } else if (header[i].contains("condition") && header[i].contains(":")) {
206         String condition = tryReplacePredefinedConditionNames(header[i]);
207         if (condition.contains(":")) {
208           condition = header[i].split(":")[1].trim();
209         }
210 
211         for (char c : condition.toCharArray()) {
212           if (Character.isUpperCase(c)) {
213             error = "Conditions are not allowed to contain upper case characters: " + condition;
214             return null;
215           }
216         }
217         if (condition.contains(" ")) {
218           error = "Conditions are not allowed to contain spaces: " + condition;
219           return null;
220         }
221         char first = condition.charAt(0);
222         if (first >= '0' && first <= '9') {
223           error = "Conditions are not allowed to start with a number: " + condition;
224           return null;
225         }
226         factors.add(i);
227       } else if (header[i].contains("Locus:")) {
228         loci.add(i);
229       } else {
230         meta.add(i);
231       }
232     }
233     // sort attributes
234     Set<Integer> entityFactors = new HashSet<Integer>();
235     Set<Integer> extractFactors = new HashSet<Integer>();
236     for (int col : factors) {
237       Map<String, String> idToVal = new HashMap<String, String>();
238       boolean ent = true;
239       boolean extr = true;
240       for (String[] row : data) {
241         String val = row[col];
242         String sourceID = row[headerMapping.get("Secondary Name")]; // for now: one row, one new
243                                                                     // source
244         // String replicateID = "";
245         // if (headerMapping.containsKey("Technical Replicates")) {
246         // replicateID = row[headerMapping.get("Technical Replicates")];
247         // }
248         String extractID = sourceID + row[headerMapping.get("Sample type")];// + replicateID; TODO
249                                                                             // might use more cols
250         // if different for same entities: not an entity attribute
251         if (idToVal.containsKey(sourceID)) {
252           if (!idToVal.get(sourceID).equals(val))
253             ent = false;
254         }
255         if (idToVal.containsKey(extractID)) {
256           if (!idToVal.get(extractID).equals(val))
257             extr = false;
258         }
259         idToVal.put(sourceID, val);
260         idToVal.put(extractID, val);
261       }
262       if (ent)
263         entityFactors.add(col);
264       if (extr)
265         extractFactors.add(col);
266     }
267     // if (parseGraph) {
268     // for (int factorCol : factors) {
269     // String label = parseXMLPartLabel(header[factorCol]);
270     // nodesForFactorPerLabel.put(label, new LinkedHashSet<SampleSummary>());
271     // }
272     // nodesForFactorPerLabel.put("None", new LinkedHashSet<SampleSummary>());
273     // }
274     // create samples
275     List<ISampleBean> beans = new ArrayList<>();
276 
277     // worst case order of samples (hierarchy):
278     // organism - tissue - protein - peptide - protein pool - peptides of pool - protein
279     // fractions/cycles - peptides of fractions/cycles - pool of fractions - peptides of pool - ms
280     // measurements
281     List<MetaboSampleHierarchy> order =
282         new ArrayList<>(Arrays.asList(MetaboSampleHierarchy.values()));
283     Map<MetaboSampleHierarchy, List<ISampleBean>> samplesInOrder = new HashMap<>();
284     for (MetaboSampleHierarchy level : order) {
285       samplesInOrder.put(level, new ArrayList<>());
286     }
287 
288     Map<String, TSVSampleBean> sourceIDToSample = new HashMap<>();
289     Map<String, TSVSampleBean> tissueToSample = new HashMap<>();
290     Map<String, TSVSampleBean> metaboliteToSample = new HashMap<>();
291     // TODO another level of measurements?
292     Map<ExtendedMSProperties, String> msPropertiesToID = new HashMap<>();
293     Map<CultureProperties, String> culturePropertiesToID = new HashMap<>();
294     Map<MetabolitePrepProperties, String> metaboPrepPropertiesToID = new HashMap<>();
295 
296     speciesSet = new HashSet<String>();
297     tissueSet = new HashSet<String>();
298     analyteSet = new HashSet<String>();
299     analyteSet.add("SMALLMOLECULES");
300     int rowID = 0;
301     int sampleID = 0;
302 
303     for (String[] row : data) {
304       fillParsedCategoriesToValuesForRow(headerMapping, row);
305       rowID++;
306       boolean special = false;
307       if (!special) {
308         for (String col : mandatoryFilled) {
309           if (row[headerMapping.get(col)].isEmpty()) {
310             error = col + " is a mandatory field, but it is not set for row " + rowID + "!";
311             return null;
312           }
313         }
314         String sourceID = row[headerMapping.get("Secondary Name")];
315         String species = row[headerMapping.get("Species")];
316         String expressionSystem = null;
317         // if (headerMapping.containsKey("Expression System")) {
318         // expressionSystem = row[headerMapping.get("Expression System")];
319         // }
320         // String tissue = row[headerMapping.get("Tissue")];//TODO whole organism?
321         String replicateID = "";
322         // if (headerMapping.containsKey("Technical Replicates")) {
323         // replicateID = row[headerMapping.get("Technical Replicates")];
324         // }
325 
326         // String fileName = row[headerMapping.get("File Name")];// TODO Sample Number?
327 
328         String sampleKey = row[headerMapping.get(SAMPLE_KEYWORD)];
329         // String sampleAltName = row[headerMapping.get(SAMPLE_ALTNAME_KEYWORD)];
330 
331         String cultureMedium = row[headerMapping.get("Medium")];
332         String harvestingConditions = row[headerMapping.get("Harvesting conditions")];
333         String washingSolvent = row[headerMapping.get("Washing solvent")];
334         String sampleSolvent = row[headerMapping.get("Sample solvent")];
335         String cultureType = row[headerMapping.get("Culture type")];
336         String sampleType = row[headerMapping.get("Sample type")];
337         String cellLysis = row[headerMapping.get("Cell lysis")];
338         String lysisParams = row[headerMapping.get("Lysis parameters")];
339         String strainCollNumber = row[headerMapping.get("Strain lab collection number")];
340 
341         speciesSet.add(species);
342         // tissueSet.add(tissue);//TODO set standard name and show user?
343 
344         // always one new measurement per row
345         sampleID++;
346         TSVSampleBean msRun = new TSVSampleBean(Integer.toString(sampleID), SampleType.Q_MS_RUN, "",
347             fillMetadata(header, row, meta, factors, new ArrayList<>(), SampleType.Q_MS_RUN));
348          msRun.addProperty("File", sampleKey);//TODO? file name
349         msRun.addProperty("Q_SAMPLE_SOLVENT", sampleSolvent);
350 
351         String lcmsMethod = row[headerMapping.get("LC MS Method Name")];
352         String msDevice = row[headerMapping.get("MS device")];
353         String lcDevice = row[headerMapping.get("LC device")];
354         String column = row[headerMapping.get("LC column name")];
355         String ionMode = row[headerMapping.get("MS ion mode")];
356         String lcDetectionMethod = row[headerMapping.get("LC detection method")];// TODO do not put
357                                                                                  // this into ms
358                                                                                  // props
359         ExtendedMSPropertiestml#ExtendedMSProperties">ExtendedMSProperties msProperties = new ExtendedMSProperties(lcmsMethod, msDevice);
360         msProperties.setLCDetectionMethod(lcDetectionMethod);
361         msProperties.setLCDevice(lcDevice);
362         msProperties.setIonizationMode(ionMode);
363         msProperties.setColumnName(column);
364         msProperties.setWashingSolvent(washingSolvent);
365         String expID = Integer.toString(msProperties.hashCode());
366         msPropertiesToID.put(msProperties, expID);
367         msRun.setExperiment(expID);
368 
369         samplesInOrder.get(MetaboSampleHierarchy.MassSpecRun).add(msRun);
370 
371         // if organism id not known => create organism entity. put in map.
372         // else get organism entity. same for tissue. this is done for pool rows, as well
373         TSVSampleBean sampleSource = sourceIDToSample.get(sourceID);
374         if (sampleSource == null) {
375           sampleID++;
376 
377           sampleSource = new TSVSampleBean(Integer.toString(sampleID),
378               SampleType.Q_BIOLOGICAL_ENTITY, sourceID,
379               fillMetadata(header, row, meta, entityFactors, loci, SampleType.Q_BIOLOGICAL_ENTITY));
380           sampleSource.addProperty("Q_STRAIN_LAB_COLLECTION_NUMBER", strainCollNumber);
381           // sampleSource.addProperty("Q_EXTERNALDB_ID", sourceID);
382           samplesInOrder.get(MetaboSampleHierarchy.Organism).add(sampleSource);
383           sourceIDToSample.put(sourceID, sampleSource);
384 
385           CulturePropertiesrties.html#CultureProperties">CultureProperties props = new CultureProperties(cultureMedium, cultureType);
386           String cultExpID = Integer.toString(props.hashCode());
387           culturePropertiesToID.put(props, cultExpID);
388           sampleSource.setExperiment(cultExpID);
389 
390           if (expressionSystem != null) {
391             speciesSet.add(expressionSystem);
392             sampleSource.addProperty("Q_EXPRESSION_SYSTEM", expressionSystem);
393           }
394         }
395         // we don't have tissue ids, so we build unique identifiers by adding sourceID and tissue
396         // name
397         // TODO use sample type instead of tissue
398         String tissueID = sourceID + "-" + sampleType + "-" + replicateID;
399         TSVSampleBean tissueSample = tissueToSample.get(tissueID);
400         if (tissueSample == null) {
401           sampleID++;
402 
403           tissueSample = new TSVSampleBean(Integer.toString(sampleID),
404               SampleType.Q_BIOLOGICAL_SAMPLE, tissueID, fillMetadata(header, row, meta,
405                   extractFactors, loci, SampleType.Q_BIOLOGICAL_SAMPLE));
406           tissueSample.addProperty("Q_PRIMARY_TISSUE", "Whole organism");
407           samplesInOrder.get(MetaboSampleHierarchy.Tissue).add(tissueSample);
408           tissueSample.addParentID(sampleSource.getCode());
409           tissueSample.addProperty("Q_EXTERNALDB_ID", tissueID);
410           tissueToSample.put(tissueID, tissueSample);
411 
412           MetabolitePrepProperties props =
413               new MetabolitePrepProperties(harvestingConditions, cellLysis, lysisParams);
414           String prepExpID = Integer.toString(props.hashCode());
415           metaboPrepPropertiesToID.put(props, prepExpID);
416           tissueSample.setExperiment(prepExpID);
417         }
418 
419         // if sample secondary name not known => create metabolite sample
420         TSVSampleBean metabolite = metaboliteToSample.get(sampleKey);
421         if (metabolite == null) {
422           sampleID++;
423 
424           metabolite = new TSVSampleBean(Integer.toString(sampleID), SampleType.Q_TEST_SAMPLE,
425               sampleKey, fillMetadata(header, row, meta, factors, loci, SampleType.Q_TEST_SAMPLE));
426           samplesInOrder.get(MetaboSampleHierarchy.Molecules).add(metabolite);
427           metabolite.addParentID(tissueSample.getCode());
428           metabolite.addProperty("Q_EXTERNALDB_ID", sampleKey);
429           metaboliteToSample.put(sampleKey, metabolite);
430           metabolite.addProperty("Q_SAMPLE_TYPE", "SMALLMOLECULES");
431         }
432         msRun.addParentID(metabolite.getCode());
433       }
434     }
435     experimentInfos = new HashMap<String, List<Map<String, Object>>>();
436 
437     // Cell cultures
438     List<Map<String, Object>> cultures = new ArrayList<Map<String, Object>>();
439     for (CultureProperties props : culturePropertiesToID.keySet()) {
440       Map<String, Object> propMap = props.getPropertyMap();
441       propMap.put("Code", Integer.toString(props.hashCode()));// used to match samples to their
442       // experiments later
443       cultures.add(propMap);
444     }
445     experimentInfos.put("Q_EXPERIMENTAL_DESIGN", cultures);
446 
447     // Extract preparation experiments
448     List<Map<String, Object>> extractPreparations = new ArrayList<Map<String, Object>>();
449     for (MetabolitePrepProperties props : metaboPrepPropertiesToID.keySet()) {
450       Map<String, Object> propMap = props.getPropertyMap();
451       propMap.put("Code", Integer.toString(props.hashCode()));// used to match samples to their
452       // experiments later
453       extractPreparations.add(propMap);
454     }
455     experimentInfos.put("Q_SAMPLE_EXTRACTION", extractPreparations);
456 
457     // MS experiments
458     List<Map<String, Object>> msExperiments = new ArrayList<Map<String, Object>>();
459     for (ExtendedMSProperties expProperties : msPropertiesToID.keySet()) {
460       Map<String, Object> propMap = expProperties.getPropertyMap();
461       propMap.put("Code", msPropertiesToID.get(expProperties));
462       msExperiments.add(propMap);
463     }
464     experimentInfos.put("Q_MS_MEASUREMENT", msExperiments);
465     for (MetaboSampleHierarchy level : order) {
466       beans.addAll(samplesInOrder.get(level));
467       // printSampleLevel(samplesInOrder.get(level));TODO
468     }
469     return beans;
470   }

472   private String tryReplacePredefinedConditionNames(String condition) {
473     Map<String, String> headNamesToConditions = new HashMap<>();
474     headNamesToConditions.put("Growth conditions: Temperature", "growth_temperature");
475     headNamesToConditions.put("Growth conditions: Time", "growth_time");
476     headNamesToConditions.put("Growth conditions: rpm", "rpm");
477     if (headNamesToConditions.containsKey(condition)) {
478       return headNamesToConditions.get(condition);
479     }
480     return condition;
481   }

528   private boolean checkUniqueIDsBetweenSets(Set<String> speciesSet, Set<String> tissueSet) {
529     Set<String> intersection1 = new HashSet<String>(speciesSet);
530     intersection1.retainAll(tissueSet);
531     if (!intersection1.isEmpty()) {
532       error = "Entry " + intersection1.iterator().next()
533           + " was found as organism and tissue. It can't be both!";
534       return false;
535     }
536     return true;
537   }

551   private HashMap<String, Object> fillMetadata(String[] header, String[] data, List<Integer> meta,
552       Set<Integer> factors, List<Integer> loci, SampleType type) {
553     Map<String, List<String>> headersToOpenbisCode = headersToTypeCodePerSampletype.get(type);
554     HashMap<String, Object> res = new HashMap<String, Object>();
555     if (headersToOpenbisCode != null) {
556       for (int i : meta) {
557         String label = header[i];
558         if (!data[i].isEmpty() && headersToOpenbisCode.containsKey(label)) {
559           for (String propertyCode : headersToOpenbisCode.get(label)) {
560             Object val = data[i];
561             // if (parsers.containsKey(propertyCode))
562             // val = parsers.get(propertyCode).parse(data[i]);
563             res.put(propertyCode, val);
564           }
565         }
566       }
567     }
568     if (factors.size() > 0)
569 
570     {
571       String fRes = "";
572       for (int i : factors) {
573         if (!data[i].isEmpty()) {
574           String values = unitCheck(data[i]);
575           fRes += parseXMLPartLabel(header[i]) + ": " + values + ";";
576         }
577       }
578       // remove trailing ";"
579       fRes = fRes.substring(0, Math.max(1, fRes.length()) - 1);
580       res.put("XML_FACTORS", fRes);
581     }
582     if (loci.size() > 0) {
583       String lRes = "";
584       for (int i : loci) {
585         if (!data[i].isEmpty()) {
586           lRes += parseXMLPartLabel(header[i]) + ": " + data[i] + ";";
587         }
588       }
589       // remove trailing ";"
590       lRes = lRes.substring(0, Math.max(1, lRes.length()) - 1);
591       res.put("XML_LOCI", lRes);
592     }
593     return res;
594   }

596   private String unitCheck(String string) {
597     String[] split = string.split(":");
598     if (split.length > 2)
599       return string.replace(":", " -");
600     if (split.length == 2) {
601       String unit = split[1].trim();
602       for (Unit u : Unit.values()) {
603         if (u.getValue().equals(unit))
604           return string;
605       }
606       return string.replace(":", " -");
607     }
608     return string;
609   }
