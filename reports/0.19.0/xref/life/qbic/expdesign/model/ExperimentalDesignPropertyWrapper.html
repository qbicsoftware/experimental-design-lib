10  import life.qbic.xml.properties.Property;
11  import life.qbic.xml.study.Qproperty;
12  
13  public class ExperimentalDesignPropertyWrapper {
14  
15    private Map<String, Map<Pair<String, String>, List<String>>> experimentalDesign;
16    private Map<String, List<Qproperty>> properties = new HashMap<String, List<Qproperty>>();
17  
18    public ExperimentalDesignPropertyWrapper(
19        Map<String, Map<Pair<String, String>, List<String>>> experimentalDesign,
20        Map<String, List<Property>> properties) {
21      super();
22      this.experimentalDesign = experimentalDesign;
23      this.properties = translateOldProperties(properties);
24    }
25  
26    private Map<String, List<Qproperty>> translateOldProperties(Map<String, List<Property>> props) {
27      Map<String, List<Qproperty>> res = new HashMap<String, List<Qproperty>>();
28      for (String id : props.keySet()) {
29        List<Qproperty> newProps = new ArrayList<Qproperty>();
30        for (Property p : props.get(id)) {
31          Qproperty newProp = null;
32          if (p.hasUnit()) {
33            newProp = new Qproperty(id, p.getLabel(), p.getValue(), p.getUnit());
34          } else {
35            newProp = new Qproperty(id, p.getLabel(), p.getValue());
36          }
37          newProps.add(newProp);
38        }
39        res.put(id, newProps);
40      }
41      return res;
42    }
43  
44    public Map<String, Map<Pair<String, String>, List<String>>> getExperimentalDesign() {
45      return experimentalDesign;
46    }
47  
48    public void setExperimentalDesign(
49        Map<String, Map<Pair<String, String>, List<String>>> experimentalDesign) {
50      this.experimentalDesign = experimentalDesign;
51    }
52  
53    public Map<String, List<Qproperty>> getProperties() {
54      return properties;
55    }
56  
57    public void setProperties(Map<String, List<Property>> properties) {
58      this.properties = translateOldProperties(properties);
59    }
60  
61  }
